#include <stdio.h>

int main() {
    int num;

    printf("Enter a number: ");
    scanf("%d", &num);

    if (num > 0) {
        printf("The number is positive.\n");
    } else if (num < 0) {
        printf("The number is negative.\n");
    } else {
        printf("The number is zero.\n");
    }

    return 0;
}

#include <stdio.h>

int main() {
    int num1, num2;

    printf("Enter the first number: ");
    scanf("%d", &num1);

    printf("Enter the second number: ");
    scanf("%d", &num2);

    if (num1 > num2) {
        printf("The greater value is %d.\n", num1);
    } else if (num2 > num1) {
        printf("The greater value is %d.\n", num2);
    } else {
        printf("Both numbers are equal.\n");
    }

    return 0;
}

    printf("Enter the second number: ");
    scanf("%d", &num2);

    sum = num1 + num2;

    if (sum % 2 == 0) {
        printf("The sum of the numbers is even.\n");
    } else {
        printf("The sum of the numbers is odd.\n");
    }

    return 0;
}
